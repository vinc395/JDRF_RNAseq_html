If no genes are known a priori not to be influenced by the covariates of interest, one can obtain a set of “in-silico empirical” negative controls, e.g., least significantly DE genes based on a first-pass DE analysis performed prior to RUVg normalization.

samples <- readRDS(here("data", "samples.rds"))

# Create model matrix
designRUV <- model.matrix(~0+group+age+gender, data = dgeList$samples)%>%
  set_colnames(str_remove(colnames(.), "group"))

contr.matrix <- makeContrasts(
   AvsB = B-A, 
   BvsC = C-B, 
   AvsC = C-A,
   levels = colnames(designRUV))

# Perform first pass DE on DGEList
topTableAB <- dgeList %>%
  estimateDisp(designRUV) %>%
  glmFit(designRUV) %>%
  glmLRT(contrast=contr.matrix[,1]) %>%
  topTags(n = Inf) %>%
  .$table %>%
  as_tibble() %>%
  dplyr::select(
        gene_id, gene_name, logFC, logCPM, PValue, FDR, everything()  
      ) %>%
   rownames_to_column("no.") %>%
  mutate(DE = FDR < 0.05)

topTableBC <- dgeList %>%
  estimateDisp(designRUV) %>%
  glmFit(designRUV) %>%
  glmLRT(contrast=contr.matrix[,2]) %>%
  topTags(n = Inf) %>%
  .$table %>%
  as_tibble() %>%
  dplyr::select(
        gene_id, gene_name, logFC, logCPM, PValue, FDR, everything()  
      ) %>%
   rownames_to_column("no.") %>%
  mutate(DE = FDR < 0.05)
  
topTableAC <- dgeList %>%

# Extract subset of low expressed genes from DE analysis to act as negative controls for RUVg procedure
empiricalAB <- topTableAB %>% 
  dplyr::arrange(desc(PValue)) %>%
  .[1:3000,] %>%
  .$gene_id

empiricalBC <- topTableBC %>% 
  dplyr::arrange(desc(PValue)) %>%
  .[1:3000,] %>%
  .$gene_id

empiricalAC <- topTableAC %>% 
  dplyr::arrange(desc(PValue)) %>%
  .[1:3000,] %>%
  .$gene_id

# Extract unique subset of low expressed genes from 3 pairwise comaprison
empirical <- c(empiricalAB, empiricalBC,empiricalAC) %>% unique()
